pwd - посмотреть где я
mkdir - создать деррикторию
touch - создать файл
cd зайти в деррикторию
cd .. выйти на уровень выше
cat > file_name - создать файл и сразу добавить в него инфу
cat file_name - показать что в файле
vim > редактировать файл 
:x - сохранить и выйти из вима
ls -la - показать все дерриктории и файлы
cp - копировать
mv - переименовать/переместить
rm - удалить
clear - очистить терминал
history - показать историю терминала

git restore - отказать последние изменения до состояния последнего коммита
git log - показать какие были коммиты без подробностей
git log -p - показать какие были коммиты с небольшими подробностями
git show - тоже что и git log но подробно
git diff - показать последние изменения
git diff --staged - показать изменения в файле уже подготовленном для коммита
